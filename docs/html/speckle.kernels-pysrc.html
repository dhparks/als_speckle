597      __global float* input, 
598      __global float* walls, 
599      __global float* poswalls, 
600      __global float* negwalls, 
601      int N) // size 
602   
603  {        
604           
605          #define compare(a,b) (1-sign(a)*sign(b))/2 
606           
607          // i and j are the x and y coordinates of the candidate pixel 
608          int i = get_global_id(0); 
609          int j = get_global_id(1); 
610           
611          // grab the 8 nearest neighbors of point (i,j). use modulo arithmetic to enforce cyclic boundary conditions. 
612          // loops have been manually unrolled for speed increase 
613          int nx1 = i-1; 
614          int nx2 = i; 
615          int nx3 = i+1; 
616          int ny1 = j-1; 
617          int ny2 = j; 
618          int ny3 = j+1; 
619           
620          if (nx1 < 0 || nx1 >= N) {nx1 = (nx1+N)%N;} 
621          if (nx2 < 0 || nx2 >= N) {nx2 = (nx2+N)%N;} 
622          if (nx3 < 0 || nx3 >= N) {nx3 = (nx3+N)%N;} 
623           
624          if (ny1 < 0 || ny1 >= N) {ny1 = (ny1+N)%N;} 
625          if (ny2 < 0 || ny2 >= N) {ny2 = (ny2+N)%N;} 
626          if (ny3 < 0 || ny3 >= N) {ny3 = (ny3+N)%N;} 
627           
628          float val11 = input[nx1+ny1*N];  
629          float val12 = input[nx1+ny2*N]; 
630          float val13 = input[nx1+ny3*N]; 
631          float val21 = input[nx2+ny1*N]; 
632          float val22 = input[nx2+ny2*N]; // this is the candidate value; the others are the neighbors 
633          float val23 = input[nx2+ny3*N]; 
634          float val31 = input[nx3+ny1*N]; 
635          float val32 = input[nx3+ny2*N]; 
636          float val33 = input[nx3+ny3*N];  
637           
638          // the function "compare" returns 0 if both inputs are the same sign and 1 if the signs differ. this is how a domain wall is detected. 
639          float preout = compare(val11,val22)+compare(val12,val22)+compare(val13,val22)+ 
640                                     compare(val21,val22)                     +compare(val23,val22)+ //skip comparing val22 to itself! 
641                                     compare(val31,val22)+compare(val32,val22)+compare(val33,val22); 
642                                    
643          // write to the output buffers 
644          float out = 0.0f; 
645          float posout = 0.0f; 
646          float negout = 0.0f; 
647           
648          if (preout > 0.1f) {out = 1.0f;} 
649          if (preout > 0.1f && val22 > 0.0f)  {posout = 1.0f;} 
650          if (preout > 0.1f && val22 <= 0.0f) {negout = 1.0f;} 
651           
652          walls[i+N*j]    = out; 
653          poswalls[i+N*j] = posout; 
654          negwalls[i+N*j] = negout; 
655           
656  }"""
